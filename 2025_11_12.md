## 2824. Count Pairs Whose Sum is Less than Target
```
class Solution:
    def countPairs(self, nums: List[int], target: int) -> int:
        """
        Time complexity = O(Nlog(N))
        Space Complexity = O(1)
        """
        nums = sorted(nums)
        print(nums)
        i = 0
        j = len(nums) - 1
        result = 0

        while i < j:
            if nums[i] + nums[j] < target:
                result += (j - i)
                i += 1
            else:
                j -= 1
                result += 0
        return result;
```

## 1351. Count Negative Numbers in a Sorted Matrix
```
class Solution:
    def fun(self, arr: List[int]) -> int:
        """
        Time Complity = O(nlog(n))
        Space Complexity = O(1)
        """
        i = 0
        j = len(arr)-1
        result = -1
        while i <= j:
            mid = int((i+j)/2)
            if arr[mid] < 0:
                result = mid
                j = mid - 1
            elif arr[mid] > 0:
                i = mid + 1
            elif arr[mid] == 0:
                i = mid + 1
        if result == -1:
            return 0
        else:
            return len(arr) - result


    def countNegatives(self, grid: List[List[int]]) -> int:
        """
        Time Complexity = O(N*log(N))
        Space Complexity = O(1)
        """
        result = 0
        for row in grid:
            result += self.fun(row)
        
        return result
```

```
    def countNegatives(self, grid: List[List[int]]) -> int:
        """
        Time Complexity = O(n+m)
        Space Complexity = O(1)
        """
        result = 0
        i = len(grid)-1
        j = 0
        while((i >= 0) & (j < len(grid[0]))):
            if grid[i][j] < 0:
                result = result + (len(grid[i]) - j)
                i = i - 1
            else:
                j = j + 1 
        
        return result
```
