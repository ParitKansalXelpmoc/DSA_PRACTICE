## 2824. Count Pairs Whose Sum is Less than Target
```
class Solution:
    def countPairs(self, nums: List[int], target: int) -> int:
        """
        Time complexity = O(Nlog(N))
        Space Complexity = O(1)
        """
        nums = sorted(nums)
        print(nums)
        i = 0
        j = len(nums) - 1
        result = 0

        while i < j:
            if nums[i] + nums[j] < target:
                result += (j - i)
                i += 1
            else:
                j -= 1
                result += 0
        return result;

## 1351. Count Negative Numbers in a Sorted Matrix
```
class Solution:
    def fun(self, arr: List[int]) -> int:
        """
        Time Complity = O(nlog(n))
        Space Complexity = O(1)
        """
        i = 0
        j = len(arr)-1
        result = -1
        while i <= j:
            mid = int((i+j)/2)
            if arr[mid] < 0:
                result = mid
                j = mid - 1
            elif arr[mid] > 0:
                i = mid + 1
            elif arr[mid] == 0:
                i = mid + 1
        if result == -1:
            return 0
        else:
            return len(arr) - result


    def countNegatives(self, grid: List[List[int]]) -> int:
        """
        Time Complexity = O(N*log(N))
        Space Complexity = O(1)
        """
        result = 0
        for row in grid:
            result += self.fun(row)
        
        return result

                i = i - 1
            else:
                j = j + 1

## 2089. Find Target Indices After Sorting Array
```
class Solution:
    def fun_upper(self, nums: List[int], target: int) -> int:
        start = 0
        end = len(nums) - 1
        result = -1

        while start <= end:
            mid = (start + end)//2
            if nums[mid] < target:
                start = mid + 1
            elif nums[mid] > target:
                end = mid - 1
            else:
                result = mid
                start = mid + 1
        return result
    
    def fun_lower(self, nums: List[int], target: int) -> int:
        start = 0
        end = len(nums) - 1
        result = -1

        while start <= end:
            mid = (start + end)//2
            if nums[mid] < target:
                start = mid + 1
            elif nums[mid] > target:
                end = mid - 1
            else:
                result = mid
                end = mid - 1
        return result
    

    def targetIndices(self, nums: List[int], target: int) -> List[int]:
        nums = sorted(nums)
        start = self.fun_lower(nums, target)
        end = self.fun_upper(nums, target)
        if (start == -1) | (end == -1):
            return []
        result = []
        for x in range(start, end+1):
            result.append(x)
        return result
```

## 2529. Maximum Count of Positive Integer and Negative Integer
```
class Solution:
    def upper_neg(self, nums: List[int]) -> int:
        result = -1
        start = 0
        end = len(nums) - 1
        while start <= end:
            mid = (start + end)//2
            if nums[mid] < 0:
                result = mid
                start = mid + 1
            elif nums[mid] >= 0:
                end = mid - 1
        return result
    
    def lower_pos(self, nums: List[int]) -> int:
        result = -1
        start = 0
        end = len(nums) - 1
        while start <= end:
            mid = (start + end)//2
            if nums[mid] < 0:
                start = mid + 1
            elif nums[mid] > 0:
                result = mid
                end = mid - 1
            else:
                start = mid + 1
        return result

    def maximumCount(self, nums: List[int]) -> int:
        temp1 = self.upper_neg(nums)
        negs = 0
        if temp1 != -1:
            negs = temp1 + 1
        temp2 = self.lower_pos(nums)
        pos = 0
        if temp2 != -1:
            pos = len(nums) - temp2
        return max(negs, pos)
```
